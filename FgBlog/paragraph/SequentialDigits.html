class Solution(object):
    def sequentialDigits(self, low, high):
        """
        :type low: int
        :type high: int
        :rtype: List[int]
        """
        res = []
        for i in range(low,high):
            a = [] 
            aa = -1
            for n in str(i):
                a.append(n)
                aa += 1
                if aa > 0:
                    b = int(a[aa]) - int(a[aa - 1])
                    if b==1:
                        continue
                    a = []
                    break
            if len(a) == len(str(i)):
                res.append(i)
        return res

class Solution(object):
    def sequentialDigits(self, low, high):
        """
        :type low: int
        :type high: int
        :rtype: List[int]
        """
        a = '123456789'
        res = []
        lenLow = len(str(low))
        lenHigh = len(str(high))
        for n in range(lenLow,lenHigh+1):
            for i in range(10):
                if i + n < 10:
                    i = int(a[i:i+n])
                    if low <=i<= high:
                        res.append(i)
        return res
